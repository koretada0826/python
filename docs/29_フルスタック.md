"use client";
import { useState } from "react";
import { useRouter } from "next/navigation";

export default function NewUser() {
  const [name, setName] = useState("");
  const router = useRouter();
  
  async function submit(e: React.FormEvent) {
    e.preventDefault();
    await fetch("http://localhost:8000/api/users", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ name }),
    });
    router.push("/users");
  }
  
  return (
    <form onSubmit={submit} className="p-8 space-y-4">
      <input
        value={name}
        onChange={(e) => setName(e.target.value)}
        className="border p-2 rounded"
      />
      <button className="bg-blue-500 text-white px-4 py-2 rounded">送信</button>
    </form>
  );
}

export default async function Home() {
  const session = await auth();
  
  if (session) {
    return (
      <>
        <p>ようこそ {session.user?.name}</p>
        <form action={async () => { "use server"; await signOut(); }}>
          <button>ログアウト</button>
        </form>
      </>
    );
  }
  
  return (
    <form action={async () => { "use server"; await signIn("google"); }}>

    </form>
  );
}

## 8. JWT で FastAPI と連携(20分)

```python
# FastAPI 側
from jose import jwt
SECRET = "shared-secret"

@app.middleware("http")
async def verify_jwt(request, call_next):
    token = request.headers.get("Authorization", "").replace("Bearer ", "")
    if token:
        try:
            payload = jwt.decode(token, SECRET, algorithms=["HS256"])
            request.state.user_id = payload["sub"]
        except: pass
    return await call_next(request)
```

```ts
// Next 側でトークンを送る
const session = await auth();
const res = await fetch("http://localhost:8000/api/me", {
  headers: { Authorization: `Bearer ${session?.token}` }
});
```

## 11. WebSocket / Server-Sent Events(20分)

### SSE(片方向の継続送信)

`src/app/api/stream/route.ts`:
```ts
export async function GET() {
  const stream = new ReadableStream({
    async start(controller) {
      for (let i = 0; i < 10; i++) {
        controller.enqueue(`data: ${i}\n\n`);
        await new Promise(r => setTimeout(r, 1000));
      }
      controller.close();
    }
  });
  return new Response(stream, {
    headers: { "Content-Type": "text/event-stream" }
  });
}
```

```tsx
"use client";
import { useEffect, useState } from "react";

export default function Stream() {
  const [msgs, setMsgs] = useState<string[]>([]);
  useEffect(() => {
    const es = new EventSource("/api/stream");
    es.onmessage = e => setMsgs(m => [...m, e.data]);
    return () => es.close();
  }, []);
  return <ul>{msgs.map((m, i) => <li key={i}>{m}</li>)}</ul>;
}
```

→ AIチャットの**逐次表示**(stream)で必須。
