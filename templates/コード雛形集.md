## 6. Webスクレイピング

```python
import requests
from bs4 import BeautifulSoup
import time

headers = {
    "User-Agent": "Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_7)"
}

urls = ["https://example.com/page1", "https://example.com/page2"]
results = []

for url in urls:
    r = requests.get(url, headers=headers)
    if r.status_code != 200:
        print(f"失敗: {url}")
        continue
    
    soup = BeautifulSoup(r.text, "lxml")
    for item in soup.select(".item"):
        title = item.select_one(".title").text.strip()
        price = item.select_one(".price").text.strip()
        results.append({"title": title, "price": price})
    
    time.sleep(1)  # マナー

import pandas as pd
pd.DataFrame(results).to_csv("scraped.csv", index=False, encoding="utf-8")
```

## 7. PDF からテキスト抽出

```python
import pdfplumber
import re

results = []
with pdfplumber.open("invoice.pdf") as pdf:
    for page in pdf.pages:
        text = page.extract_text()
        # 正規表現で項目抽出
        m = re.search(r"請求番号:\s*(\S+)", text)
        if m:
            results.append({"番号": m.group(1)})
```

## 8. メール送信(Gmail)

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import os
from dotenv import load_dotenv

load_dotenv()

def send_mail(to, subject, body):
    msg = MIMEMultipart()
    msg["From"] = os.getenv("GMAIL_USER")
    msg["To"] = to
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "plain"))
    
    with smtplib.SMTP_SSL("smtp.gmail.com", 465) as server:
        server.login(os.getenv("GMAIL_USER"), os.getenv("GMAIL_PASS"))
        server.send_message(msg)

# 一斉送信
import pandas as pd
df = pd.read_csv("customers.csv")
for _, row in df.iterrows():
    send_mail(
        to=row["メアド"],
        subject="ご案内",
        body=f"{row['名前']} 様\n\nお世話になっております。"
    )
```

## 9. Claude API

```python
import os
from anthropic import Anthropic
from dotenv import load_dotenv

load_dotenv()
client = Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))

def ask(prompt):
    response = client.messages.create(
        model="claude-opus-4-7",
        max_tokens=1024,
        messages=[{"role": "user", "content": prompt}]
    )
    return response.content[0].text

answer = ask("Pythonとは?")
print(answer)
```

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def ask(prompt):
    response = client.chat.completions.create(
        model="gpt-4o",

    )

## 12. 定期実行(schedule)

```python
import schedule
import time

def 朝のタスク():
    print("おはよう!")
    # 何か処理

schedule.every().day.at("06:00").do(朝のタスク)
schedule.every().monday.at("09:00").do(朝のタスク)
schedule.every(10).minutes.do(朝のタスク)

while True:
    schedule.run_pending()
    time.sleep(60)
```
